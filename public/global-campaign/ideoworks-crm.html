Follow these steps once. After setup, every team member who opens this file sees the same live data — no exports needed.
1
Create a new Google Sheet
Go to sheets.google.com and create a blank spreadsheet. Name it anything — e.g. "IDEOWORKS CRM Data". You only need one Sheet tab, leave it as "Sheet1".
2
Open Apps Script
Inside your Google Sheet, click Extensions → Apps Script. Delete any default code in the editor. Paste the code below exactly as-is.
const SHEET_NAME = 'Sheet1'; const HEADERS = ['id','firstName','lastName','company','email','phone','country', 'source','stage','value','notes','nextFollowUp','campaign', 'serviceInterest','budgetRange','metaId','createdAt','updatedAt']; // ── WhatsApp intake form HTML ───────────── function getIntakeFormHtml_() { const url = ScriptApp.getService().getUrl(); return ` IDEOWORKS — Contact Us
IDEOWORKS

Tell us about your project

Fill in your details and our team will reach out within 24 hours.

✅

We received your message!

Our team will contact you within 24 hours.

`; } function setupSheet_() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) sheet = ss.insertSheet(SHEET_NAME); if (!sheet.getLastColumn() || sheet.getRange(1,1).getValue() !== 'id') { sheet.getRange(1,1,1,HEADERS.length).setValues([HEADERS]); sheet.setFrozenRows(1); sheet.getRange(1,1,1,HEADERS.length).setFontWeight('bold') .setBackground('#1C1B19').setFontColor('#ffffff'); } return sheet; } function doGet(e) { // Serve intake form when ?form=1 if (e && e.parameter && e.parameter.form === '1') { return HtmlService.createHtmlOutput(getIntakeFormHtml_()) .setTitle('IDEOWORKS — Contact Us') .addMetaTag('viewport','width=device-width,initial-scale=1'); } const sheet = setupSheet_(); const last = sheet.getLastRow(); let leads = []; if (last > 1) { const data = sheet.getRange(2,1,last-1,HEADERS.length).getValues(); leads = data .map(row => { const o={}; HEADERS.forEach((h,i)=>o[h]=String(row[i]||'')); return o; }) .filter(l => l.id); } return ContentService .createTextOutput(JSON.stringify({ok:true, leads})) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { const body = JSON.parse(e.postData.contents); const sheet = setupSheet_(); // Single lead from intake form if (body.action === 'addLead' && body.lead) { const row = HEADERS.map(h => String(body.lead[h]||'')); sheet.appendRow(row); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } if (body.action === 'saveAll') { const last = sheet.getLastRow(); if (last > 1) sheet.deleteRows(2, last - 1); if (body.leads && body.leads.length) { const rows = body.leads.map(l => HEADERS.map(h => String(l[h]||''))); sheet.getRange(2,1,rows.length,HEADERS.length).setValues(rows); Deploy as a Web App
